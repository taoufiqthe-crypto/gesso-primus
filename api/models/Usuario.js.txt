// models/Usuario.js
const { DataTypes } = require("sequelize");
const sequelize = require("../sequelize");

const Usuario = sequelize.define("Usuario", {
  nome: {
    type: DataTypes.STRING,
    allowNull: false
  },
  email: {
    type: DataTypes.STRING,
    allowNull: false,
    unique: true
  },
  senha: {
    type: DataTypes.STRING,
    allowNull: false
  }
}, {
  tableName: "usuarios"
});

module.exports = Usuario;
